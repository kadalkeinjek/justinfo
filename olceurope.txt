    function list_all_shipping_countries(){
        var is_open = $("#list_all_shipping_country").css('display');
        if(is_open == 'block'){
            $("#list_all_shipping_country").css('display','none');
        }else{
            var countryData = $("#list_all_shipping_country").html();

            if(countryData == '')

                    url: 'https://olceurope.com/',
                    data: 'action=show',

                    success: function(response) {
                        $("#list_all_shipping_country").html(response);
                        $("#list_all_shipping_country").css('display','block');
                    }

            }
            else
            {
                $("#list_all_shipping_country").css('display','block');
            }
        }
        return false;
    }

            $(document).ready(function() {
                let hoverTimeout;
                let menuLoaded = false; // Flag to check if the menu is already loaded
                const staticMenuDiv = $('.static_header_menu');
    
                // Handle hover event with delay for the entire static menu div
                staticMenuDiv.on('mouseenter', function() {
                    // Check if the menu has already been loaded
                    if (menuLoaded) {
                        return; // Exit the function if the menu has already been loaded
                    }
    
                    // Clear any previous hover timeout to avoid multiple AJAX requests
                    clearTimeout(hoverTimeout);
    
                    // Set a delay before sending the AJAX request
                    hoverTimeout = setTimeout(function() {
                        // Send the AJAX request to load the full menu section
                        $.ajax({
                            url: UKSS_LIVE_HOST + '/index.php?main_page=ajax_header_menu_section',
                            method: 'POST',
                            data: {
                                load_menu_header: 'yes'
                            }, // Send the required POST parameter
                            success: function(response) {
                                // Replace the static menu with the dynamic menu content received from the server
                                staticMenuDiv.html(response);
                                // Set the flag to true, indicating that the menu has been loaded
                                menuLoaded = true;
                            },
                            error: function() {
                                console.error('Failed to load the full menu.');
                            }
                        });
                    }, 5); // Delay of 10ms before making the request
                }).on('mouseleave', function() {
                    // Clear the timeout if the user quickly leaves the menu area to avoid sending the AJAX request
                    clearTimeout(hoverTimeout);
                });
            });

            function removeKeysForProduct(formData, productId, sizeid) {
                // Construct the key we want to keep
                var keyToKeep = 'id[' + productId + '][' + sizeid + ']';
    
                // Iterate over formData and remove unwanted keys
                for (var key in formData) {
                    // Check if the key is for the specified product ID and is not the key we want to keep
                    if (key.startsWith('id[' + productId + ']') && key !== keyToKeep) {
                        // Check if the key exists in formData and its value is not an empty string
                        if (formData[key] !== undefined && formData[key] !== "") {
                            formData[key] = "";  // Clear the value
                        } else {
                            delete formData[key]; // Remove the key
                        }
                    }
                }
    
                return formData;
            }
    
                    function remove_customization(offset, productId, sizeid){
                if ($(".cart-message-success").length) {
                    // Show the success message by changing its display property to 'block'

                }
                $(".sidebar-cart__loader").addClass("active");
    
                // Get the serialized form data
                var formData = $("#basket").serializeArray();
    
                // Convert form data to an object
                var dataObj = {};
                formData.forEach(function(item) {
                    // Preserve the first occurrence of keys with numbers (e.g., main_page0, action0, etc.)
                    // Allow array-style keys (cart_quantity[] and products_id[]) to be stored as arrays
                    if (item.name.endsWith('[]')) {
                        if (!dataObj[item.name]) {
                            dataObj[item.name] = [];
                        }
                        dataObj[item.name].push(item.value); // Store multiple values for array-style keys
                    } else if (!dataObj.hasOwnProperty(item.name)) {
                        // If the key does not exist yet, add it
                        dataObj[item.name] = item.value;
                    }
                    // Otherwise, ignore duplicate keys (they won't overwrite the first occurrence)
                });
    
                // Filter the form data to keep only the necessary keys for the specified offset
                dataObj = removeKeysForProduct(dataObj , productId, sizeid);
    
                // Send the AJAX request with the filtered data
                $.ajax({
                    type: "POST",

        } else {

        if (applied_filters) {
            newUrl.searchParams.set('filter', applied_filters);
        } else {
            newUrl.searchParams.delete('filter');
        }

    function getAppliedFiltersFromUrl() {
        //Step 1: Extract the query string from the URL and parse the URL parameters.
        const url = window.location.href;
    
        // Extract the query string from the URL
        const queryString = url.split('?')[1];
        if (!queryString) {
            return {}; // Return empty if no query string exists
        }
    
        // Parse the URL parameters
        const urlParams = new URLSearchParams(queryString);
        //Step 2: Extract the filter parameter from the URL query string.
        const filterParam = urlParams.get('filter');
        
        // If filterParam is null or empty, return an empty object
        if (!filterParam) {
            return {};
        }
        
        //Step 3: Decode and sanitize the filter parameter and split the filters by `&&` and trim each part.
        const decodedFilter = decodeURIComponent(filterParam).trim();
    
        // Split the filters by `&&` and trim each part
        const selectedFilters = decodedFilter.split("&&").map(filter => filter.trim());
    
        const appliedFilters = {};
        const regex = /([a-zA-Z_]+):\s*`([^`]+)`/g; // Added the global flag
    
        //Step 4: Process each filter and extract the field name and value.
        selectedFilters.forEach(filter => {
            // Remove parentheses and extra spaces
            filter = filter.replace(/^\(|\)$/g, '').trim();
    
            const matches = [...filter.matchAll(regex)];
    
            matches.forEach(match => {
                const key = match[1]; // Field name
                const value = match[2]; // Field value
    
                //Step 5: Add the value to the appliedFilters object
                if (!appliedFilters[key]) {
                    appliedFilters[key] = [];
                }
                if (!appliedFilters[key].includes(value)) {
                    appliedFilters[key].push(value);
                }
            });
        });
    
        //return the applied filters object
        return appliedFilters;
    }

        // Map of special colors with their hex codes
        const specialColorsHexMap = {
            'none': '#FFFFFF',
            'red': '#FF0000',
            'orange': '#FFA500',

            'navy': '#000080',
            'black': '#000000',

            'sky blue': '#87CEEB',
            'sky': '#87CEEB',
            'cardinal red': '#C41E3A',
            'burgundy': '#800020',
            'lime punch': '#DFFF00',
            'fire red': '#B22222',

            'dark green': '#013220',
            'universal': '#3A3B3C',
            'cream': '#FFFDD0',
            'brown': '#A52A2A'
        };
    
        let attrStr = '';
    
        // Check if the option contains multiple colors (separated by '/' or '-')
        if (optionsValue.includes('/') || optionsValue.includes('-')) {
            let styleColors = [];
    
            // Split the option value into individual colors
            if (optionsValue.includes('/')) {
                styleColors = optionsValue.split('/');
            } else if (optionsValue.includes('-')) {
                styleColors = optionsValue.split('-');
            }
    
            // Generate a linear gradient for 2 or 3 colors
            if (styleColors.length === 2) {
                const [color1, color2] = styleColors;
                if (specialColorsHexMap[color1] && specialColorsHexMap[color2]) {
                    attrStr = `<small style="background: linear-gradient(to right, 
                        ${specialColorsHexMap[color1]} 50%, ${specialColorsHexMap[color2]} 50%);"></small>`;
                } else {
                    attrStr = `<small style="background-color: white;"></small>`;
                }
            } else if (styleColors.length === 3) {
                const [color1, color2, color3] = styleColors;
                if (
                    specialColorsHexMap[color1] &&
                    specialColorsHexMap[color2] &&
                    specialColorsHexMap[color3]
                ) {

            if (specialColorsHexMap[styleColor]) {

            } else {
                attrStr = `<small style="background-color: white;"></small>`;
            }
        }
    
        return attrStr;
    }

    // Function to fetch the currently applied filter in the Mobile device and convert those selected filter into the Parameter format accepted by BraveOtter
    function generateMobileFilterString() {
        const filters = document.querySelectorAll('.mf-filters-option-list > li');
        let filterString = '';
    
        filters.forEach((filter) => {
            const fieldName = filter.getAttribute('data-filter-content');
            const checkedOptions = filter.querySelectorAll('.mf-checkbox:checked');
            
            if (checkedOptions.length > 0) {
                let optionsString = Array.from(checkedOptions)
                    .map((checkbox) => {
                        // Retrieve the associated label element using the 'for' attribute of the checkbox
                        const labelId = checkbox.getAttribute('id');
    
                        if (labelId.indexOf("np_color_") !== -1) {
                            label = labelId.match(/np_color_(.*)/)[1]; 
                                                    return `np_color:\`${label}\``;
                                                } else {
                            label = document.querySelector(`label[for="${labelId}"]`);
                            // Get the text content of the label (the option value)
                            return `${fieldName}:\`${label.firstChild.textContent.trim()}\``; // Only get the value, not the count
                        }
                            
                        
                    })
                    .join(' || ');
    
                if (filterString) {
                    filterString += ' && ';
                }
                filterString += `(${optionsString})`;
            }
        });
    
        return filterString;
    }
    // Function to fetch the currently applied filter in the Desktop device and convert those selected filter into the Parameter format accepted by BraveOtter
    function generateDesktopFilterString(){
    
        const filterFieldsName = document.querySelectorAll('input[type="hidden"].filter_field_name');
        let filterQuery_desktop = '';
    
        filterFieldsName.forEach(fieldName => {
            const spanName = fieldName.value.trim();
            const checkboxClassName = `${spanName}_checkbox`;
            const checkboxes = document.querySelectorAll(`input[type="checkbox"].${checkboxClassName}:checked`);
            const checkedValues = Array.from(checkboxes).map(checkbox => `\`${checkbox.value}\``);  // Add backticks around the values
    
            if (checkedValues.length > 0) {
                const filterCondition = checkedValues.map(value => `${spanName}:${value}`).join(' || ');
                if (filterQuery_desktop) {
                    filterQuery_desktop += ` && (${filterCondition})`;
                } else {
                    filterQuery_desktop += `(${filterCondition})`;
                }
            }
        });
    
        return filterQuery_desktop;
    }

    function getFeaturedProducts(condition,param_type,sort_by,category_id,request_type) {
        return new Promise((resolve, reject) => {
            let req_params = '';
            if(request_type !="" && request_type == 'player_category'){
                if(category_id != ""){
                    req_params += '&category_id=' + category_id;
                }
                if(condition != "")
                {
                    req_params += '&filter=' + condition;
                }
                if(sort_by != "")
                {
                    req_params += '&sort=' + sort_by;
                }
                req_params += '&action=top_featured_products&page_type=category_page';
            }else{
                // If filter need to be aplied then I send it in Filter parameter otherwise in sort paramter 
                // This is done because for New Arrivals, Bestsellers, and Lovest Price, we apply sortiung and in case of Under 30 GBP and Hero shirt we apply filter 
                if(param_type == 'filter'){
                    req_params += '&filter=' + condition;
                    req_params += '&param_type='+param_type;
                }else{
                    req_params += '&sort=' + condition;
                    req_params += '&param_type='+param_type;
                }
                
                req_params += '&action=top_featured_products';
                //BOC: Made changes to show the product of specific landing page. BY Aditya Verma ON 28-November-2024. PMS: 156280. 
                let landing_page_condition = document.getElementById('landing_Page_applied_filter').value;
                req_params += '&landing_condition='+encodeURIComponent(landing_page_condition); 
            }
            //EOC: Made changes to show the product of specific landing page. BY Aditya Verma ON 28-November-2024. PMS: 156280. 
            $.ajax({
                url: 'https://www.uksoccershop.com/index.php?main_page=category_page_braveotter',
                type: 'POST',
                data: req_params,
                async: true, // Keep it asynchronous
                beforeSend: function() {
                    // Show the loader
                    $('.top_featured_products').html('');
                    $('.top_featured_products').html('<div style="height: 30px"><div class="loader"></div></div>');
                },
                success: function(response) {
                    const parsedResponse = JSON.parse(response);
                    if (parsedResponse.html) {
                        const htmlContent = parsedResponse.html.replace(/\\n|\\r|\n|\r/g, '');
                        resolve(htmlContent); // Resolve the Promise with the HTML
                    } else {
                        reject("No HTML content found in the response.");
                    }
                },
                error: function(error) {
                    reject(error); // Reject the Promise on error
                }
            });
        });
    }
